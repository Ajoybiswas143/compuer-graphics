def midpoint_line(x1, y1, x2, y2):
    """
    Draws a line using the Midpoint (Bresenham's) algorithm.
    This basic version handles slopes between 0 and 1.
    """
    points = []
    dx = x2 - x1
    dy = y2 - y1

    # Initial decision parameter
    d = 2 * dy - dx
    x, y = x1, y1

    while x <= x2:
        points.append((x, y))
        if d < 0:
            # Midpoint is above the line, choose East
            d = d + 2 * dy
        else:
            # Midpoint is below or on the line, choose North-East
            d = d + 2 * (dy - dx)
            y += 1
        x += 1
            
    return points

# Example usage:
line_points = midpoint_line(2, 3, 12, 8)
print("Midpoint Line Points:", line_points)
